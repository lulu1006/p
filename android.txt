MainActivity.kt
package com.example.sqlitetutorial

import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.view.View
import android.widget.EditText
import android.widget.LinearLayout
import android.widget.TextView
import android.database.sqlite.SQLiteDatabase
import android.database.sqlite.SQLiteOpenHelper
import androidx.activity.ComponentActivity


class MainActivity : ComponentActivity() {

    private lateinit var dbHelper: UsersDBHelper

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        dbHelper = UsersDBHelper(this)
    }

    fun addUser(view: View) {
        val userid = findViewById<EditText>(R.id.edittext_userid).text.toString()
        val name = findViewById<EditText>(R.id.edittext_name).text.toString()
        val age = findViewById<EditText>(R.id.edittext_age).text.toString()

        val result = dbHelper.insertUser(UserModel(userid, name, age))
        findViewById<TextView>(R.id.textview_result).text = "Added user: $result"
        findViewById<LinearLayout>(R.id.ll_entries).removeAllViews()
    }

    fun deleteUser(view: View) {
        val userid = findViewById<EditText>(R.id.edittext_userid).text.toString()
        val result = dbHelper.deleteUser(userid)
        findViewById<TextView>(R.id.textview_result).text = "Deleted user: $result"
        findViewById<LinearLayout>(R.id.ll_entries).removeAllViews()
    }

    fun showAllUsers(view: View) {
        val users = dbHelper.readAllUsers()
        val layout = findViewById<LinearLayout>(R.id.ll_entries)
        layout.removeAllViews()

        users.forEach {
            val tv = TextView(this)
            tv.textSize = 18f
            tv.text = "${it.name} - ${it.age}"
            layout.addView(tv)
        }

        findViewById<TextView>(R.id.textview_result).text =
            "Fetched ${users.size} users"
    }
}

package com.example.sqlitetutorial

import android.provider.BaseColumns

object DBContract {

    class UserEntry : BaseColumns {
        companion object {
            const val TABLE_NAME = "users"
            const val COLUMN_USER_ID = "userid"
            const val COLUMN_NAME = "name"
            const val COLUMN_AGE = "age"
        }
    }
}

UserDBHelper.kt
package com.example.sqlitetutorial

import android.content.ContentValues
import android.content.Context
import android.database.sqlite.SQLiteDatabase
import android.database.sqlite.SQLiteOpenHelper

class UsersDBHelper(context: Context) :
    SQLiteOpenHelper(context, DATABASE_NAME, null, DATABASE_VERSION) {

    override fun onCreate(db: SQLiteDatabase) {
        db.execSQL(SQL_CREATE_ENTRIES)
    }

    override fun onUpgrade(db: SQLiteDatabase, oldVersion: Int, newVersion: Int) {
        db.execSQL(SQL_DELETE_ENTRIES)
        onCreate(db)
    }

    fun insertUser(user: UserModel): Boolean {
        val db = writableDatabase
        val values = ContentValues().apply {
            put(DBContract.UserEntry.COLUMN_USER_ID,user.userId)
            put(DBContract.UserEntry.COLUMN_NAME, user.name)
            put(DBContract.UserEntry.COLUMN_AGE, user.age)
        }
        return db.insert(DBContract.UserEntry.TABLE_NAME, null, values) != -1L
    }

    fun deleteUser(userid: String): Boolean {
        val db = writableDatabase
        val rows = db.delete(
            DBContract.UserEntry.TABLE_NAME,
            "${DBContract.UserEntry.COLUMN_USER_ID}=?",
            arrayOf(userid)
        )
        return rows > 0
    }

    fun readAllUsers(): ArrayList<UserModel> {
        val users = ArrayList<UserModel>()
        val db = readableDatabase
        val cursor = db.rawQuery("SELECT * FROM ${DBContract.UserEntry.TABLE_NAME}", null)

        cursor.use {
            while (it.moveToNext()) {
                users.add(
                    UserModel(
                        it.getString(it.getColumnIndexOrThrow(DBContract.UserEntry.COLUMN_USER_ID)),
                        it.getString(it.getColumnIndexOrThrow(DBContract.UserEntry.COLUMN_NAME)),
                        it.getString(it.getColumnIndexOrThrow(DBContract.UserEntry.COLUMN_AGE))
                    )
                )
            }
        }
        return users
    }

    companion object {
        const val DATABASE_VERSION = 1
        const val DATABASE_NAME = "Users.db"

        const val SQL_CREATE_ENTRIES =
            "CREATE TABLE users (userid TEXT PRIMARY KEY, name TEXT, age TEXT)"

        const val SQL_DELETE_ENTRIES =
            "DROP TABLE IF EXISTS users"
    }
}
